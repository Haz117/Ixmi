// REGLAS DE FIRESTORE PARA DEBUGGEAR
// Ve a Firebase Console > Firestore Database > Rules
// Copia y pega estas reglas TEMPORALMENTE para diagnosticar

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // REGLAS TEMPORALES MUY PERMISIVAS - SOLO PARA DEBUGGEAR
    match /{document=**} {
      allow read, write: if true;
    }
  }
}
